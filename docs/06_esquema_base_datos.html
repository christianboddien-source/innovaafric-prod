0. Arquitectura de Datos — Visión General

INNOVAAFRIC utiliza dos bases de datos PostgreSQL complementarias:

Base de DatosPlataformaAccesoPropósito
Supabase SBSupabase.com (managed PG)Frontend JS SDK + pg_net triggersDatos accesibles desde el navegador, Storage KYC, Realtime, 56+ tablas
Railway PostgreSQL RWRailway.app (managed PG 15)Backend API exclusivamente via PrismaDatos transaccionales del backend, modelos Prisma, 7 migraciones aplicadas
Sincronización entre bases de datos
Los datos de usuarios se sincronizan entre Supabase y Railway mediante el trigger pg_net: al insertar un usuario en Supabase, el trigger dispara un webhook HTTP a POST /v1/admin/sync-user en Railway, que crea el registro equivalente en el modelo Prisma User.

2. Modelos Prisma — Railway PostgreSQL RW

El schema de Prisma define los modelos que mapean con las tablas de Railway PostgreSQL. 7 migraciones han sido aplicadas. Los modelos principales son:

// schema.prisma — modelos principales

model User {
  id            String    @id @default(uuid())
  email         String    @unique
  name          String
  phone         String?
  country       String?
  role          String    @default("user")
  level         Int       @default(1)   // 1-4
  kycStatus     String    @default("pending")
  passwordHash  String?
  referralCode  String?   @unique
  isBanned      Boolean   @default(false)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  wallet        Wallet?
  transactions  Transaction[]
  loans         Loan[]
  orders        Order[]
  apiKeys       ApiKey[]
  referrals     Referral[]
  payrollItems  PayrollItem[]
  loyaltyAcc    LoyaltyAccount?
  businessAcc   BusinessAccount?
  virtualCards  VirtualCard[]
  savingsAccs   SavingsAccount[]
  tontines      TontineMember[]
}

model Wallet {
  id          String  @id @default(uuid())
  userId      String  @unique
  user        User    @relation(fields:[userId], references:[id])
  balanceEur  Decimal @default(0) @db.Decimal(18,4)
  balanceUsd  Decimal @default(0) @db.Decimal(18,4)
  balanceGbp  Decimal @default(0) @db.Decimal(18,4)
  balanceXaf  Decimal @default(0) @db.Decimal(18,2)
  balanceXof  Decimal @default(0) @db.Decimal(18,2)
  balanceGnf  Decimal @default(0) @db.Decimal(18,2)
  createdAt   DateTime @default(now())
}

model Transaction {
  id           String   @id @default(uuid())
  userId       String
  user         User     @relation(fields:[userId], references:[id])
  type         String
  amount       Decimal  @db.Decimal(18,4)
  currency     String
  fromCurr     String?
  toCurr       String?
  exchangeRate Decimal? @db.Decimal(18,6)
  fee          Decimal  @default(0) @db.Decimal(18,4)
  status       String   @default("pending")
  reference    String?  @unique
  createdAt    DateTime @default(now())
}

model Loan {
  id              String   @id @default(uuid())
  userId          String
  user            User     @relation(fields:[userId], references:[id])
  amount          Decimal  @db.Decimal(18,4)
  currency        String
  interestRate    Decimal  @db.Decimal(5,2)
  durationMonths  Int
  monthlyPayment  Decimal  @db.Decimal(18,4)
  purpose         String?
  status          String   @default("pending")
  dueDate         DateTime?
  createdAt       DateTime @default(now())
}

model SavingsAccount {
  id          String   @id @default(uuid())
  userId      String
  user        User     @relation(fields:[userId], references:[id])
  balance     Decimal  @default(0) @db.Decimal(18,4)
  currency    String
  interestRate Decimal @db.Decimal(5,2)
  maturityDate DateTime?
  status      String   @default("active")
  createdAt   DateTime @default(now())
}

model Tontine {
  id          String   @id @default(uuid())
  name        String
  currency    String
  amount      Decimal  @db.Decimal(18,4)
  frequency   String   // weekly | monthly
  status      String   @default("active")
  createdAt   DateTime @default(now())
  members     TontineMember[]
}

model VirtualCard {
  id          String   @id @default(uuid())
  userId      String
  user        User     @relation(fields:[userId], references:[id])
  cardNumber  String   @unique
  expiryMonth Int
  expiryYear  Int
  cvv         String
  status      String   @default("active")
  balance     Decimal  @default(0) @db.Decimal(18,4)
  currency    String
  createdAt   DateTime @default(now())
}

model LoyaltyAccount {
  id          String   @id @default(uuid())
  userId      String   @unique
  user        User     @relation(fields:[userId], references:[id])
  points      Int      @default(0)
  tier        String   @default("bronze") // bronze|silver|gold|platinum
  updatedAt   DateTime @updatedAt
}

model BusinessAccount {
  id           String   @id @default(uuid())
  userId       String   @unique
  user         User     @relation(fields:[userId], references:[id])
  businessName String
  taxId        String?
  country      String
  status       String   @default("pending")
  createdAt    DateTime @default(now())
}

model Referral {
  id           String   @id @default(uuid())
  referrerId   String
  referrer     User     @relation(fields:[referrerId], references:[id])
  referredEmail String
  status       String   @default("pending") // pending|registered|rewarded
  reward       Decimal? @db.Decimal(18,4)
  createdAt    DateTime @default(now())
}

model PayrollRun {
  id        String   @id @default(uuid())
  period    String   // "2026-06"
  status    String   @default("draft") // draft|approved|paid
  totalAmount Decimal @db.Decimal(18,4)
  currency  String
  approvedBy String?
  createdAt DateTime @default(now())
  items     PayrollItem[]
}

model ApiKey {
  id          String   @id @default(uuid())
  userId      String
  user        User     @relation(fields:[userId], references:[id])
  partnerName String
  keyPrefix   String
  fullKey     String   // bcrypt hash
  environment String   @default("production")
  permissions Json     @default("[]")
  rateLimit   Int      @default(100)
  callCount   Int      @default(0)
  status      String   @default("active")
  lastUsedAt  DateTime?
  createdAt   DateTime @default(now())
}

Nota sobre RLS (Row Level Security)
RLS está deshabilitado en todas las tablas de Supabase. La seguridad de acceso a datos se gestiona completamente a nivel de aplicación: el backend Railway verifica JWT y niveles de acceso antes de cualquier operación. Considerar habilitar RLS progresivamente en producción para tablas de alta sensibilidad (wallets, kyc_requests, transactions).
